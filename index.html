function calculate() { const rebateInput = document.getElementById("rebateDate").value.trim(); const tier = parseInt(document.getElementById("tier").value); const purchasesRaw = document.getElementById("purchaseDates").value.trim(); if (!rebateInput || !purchasesRaw) return alert("Please enter all fields."); let rebateDate = new Date(rebateInput.replace(" ", "T")); if (isNaN(rebateDate)) return alert("Invalid rebate date format."); const periodDays = tier === 2 ? 7 : 9; const purchaseLines = purchasesRaw.split(/\n+/).map(line => line.trim()).filter(line => line); const purchaseDates = purchaseLines.map(line => new Date(line.replace(" ", "T"))).filter(dt => !isNaN(dt)); if (purchaseDates.length === 0) return alert("No valid purchase dates."); purchaseDates.sort((a,b) => a - b); const maxPurchaseDate = purchaseDates[purchaseDates.length - 1]; let rangeStart = new Date(rebateDate); const results = []; function getDayKey(date) { const totalSeconds = date.getHours()*3600 + date.getMinutes()*60 + date.getSeconds(); const cutoffSeconds = 8*3600; // 08:00:00 let adjustedDate = new Date(date); if (totalSeconds < cutoffSeconds) adjustedDate.setDate(adjustedDate.getDate() - 1); return adjustedDate.getFullYear() + '-' + (adjustedDate.getMonth()+1).toString().padStart(2,'0') + '-' + adjustedDate.getDate().toString().padStart(2,'0'); } while (rangeStart <= maxPurchaseDate) { // Range end: periodDays later at 08:00:00 let rangeEnd = new Date(rangeStart); rangeEnd.setDate(rangeEnd.getDate() + periodDays); rangeEnd.setHours(8,0,0,0); const dayCounts = {}; for (let dt of purchaseDates) { if (dt >= rangeStart && dt < rangeEnd) { const dayStr = getDayKey(dt); dayCounts[dayStr] = 1; // count **unique day** only } } const totalPurchases = Object.keys(dayCounts).length; results.push({ start: new Date(rangeStart), end: new Date(rangeEnd), uniqueDays: Object.keys(dayCounts).sort(), totalPurchases }); // Next range starts at previous range end rangeStart = new Date(rangeEnd); } // Build summary table let tableHTML = ''; results.forEach(r => { const highlightClass = r.totalPurchases > 0 ? 'valid-range' : ''; tableHTML += ``; }); tableHTML += '
Range StartRange End (08:00:00)Unique DaysTotal Unique Days
${r.start.toLocaleString()} ${r.end.getFullYear()}-${(r.end.getMonth()+1).toString().padStart(2,'0')}-${r.end.getDate().toString().padStart(2,'0')} 08:00:00 ${r.uniqueDays.join('
') || '-'}		 ${r.totalPurchases}
'; document.getElementById("result").innerHTML = tableHTML; }
